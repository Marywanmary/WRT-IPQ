2. 编译方案分析
方案	描述	优势	劣势	评分
方案A	单工作流多设备	管理简单，代码复用高	编译时间长，容易超时	7/10
方案B	多工作流并行	并行编译，速度快	管理复杂，资源占用高	8.5/10
方案C	矩阵策略单工作流	平衡管理与效率	配置复杂	8/10
推荐方案B: 多工作流并行，可以充分利用GitHub Actions的并行能力，提高编译效率。

3. 核心功能实现
硬盘空间监控（每15分钟检查一次）
错误处理与日志收集
中文注释支持
NSS驱动集成
多版本固件生成（Lite/Full）

YAML
# .github/workflows/ipq60xx-immortalwrt.yml
name: 编译IPQ60XX设备固件 (ImmortalWrt)

on:
  workflow_dispatch:
    inputs:
      version_suffix:
        description: '版本后缀'
        required: false
        default: ''
      clean_build:
        description: '清理编译环境'
        type: boolean
        default: false

env:
  REPO_URL: https://github.com/laipeng668/immortalwrt.git
  REPO_BRANCH: master
  CONFIG_FILE: configs/ipq60xx-immortalwrt.config
  DEVICE_NAMES: jdcloud_re-ss-01,jdcloud_re-cs-02
  FEEDS_CONF: feeds.conf.default

jobs:
  build:
    runs-on: ubuntu-22.04
    steps:
      - name: 初始化编译环境 (Initialize build environment)
        uses: actions/checkout@v4

      - name: 检查磁盘空间 (Check disk space)
        run: |
          echo "剩余磁盘空间:"
          df -h
          bash scripts/disk-monitor.sh &

      - name: 安装编译依赖 (Install build dependencies)
        run: |
          sudo apt update
          sudo apt install -y build-essential clang flex bison g++ gawk \
            gcc-multilib g++-multilib gettext git libncurses5-dev libssl-dev \
            python3-setuptools rsync swig unzip zlib1g-dev file wget curl \
            libelf-dev ecj fastjar java-propose-classpath

      - name: 下载源码 (Download source code)
        run: |
          git clone $REPO_URL -b $REPO_BRANCH openwrt
          cd openwrt
          ./scripts/feeds update -a
          ./scripts/feeds install -a

      - name: 添加自定义软件源 (Add custom feeds)
        run: |
          cd openwrt
          echo "src-git small https://github.com/kenzok8/small-package" >> feeds.conf.default
          echo "src-git sirpdboy https://github.com/sirpdboy/luci-app-netspeedtest" >> feeds.conf.default
          echo "src-git nikki https://github.com/nikkinikki-org/OpenWrt-nikki" >> feeds.conf.default
          ./scripts/feeds update -a
          ./scripts/feeds install -a

      - name: 配置编译选项 (Configure build options)
        run: |
          cd openwrt
          cp ../$CONFIG_FILE .config
          make defconfig

      - name: 下载编译依赖包 (Download build dependencies)
        run: |
          cd openwrt
          make download -j8 || make download -j1 V=s

      - name: 编译固件 (Build firmware)
        run: |
          cd openwrt
          # 编译Lite版本
          echo "CONFIG_TARGET_DEVICE_qualcommax_ipq60xx_DEVICE_jdcloud_re-ss-01_lite=y" >> .config
          echo "CONFIG_TARGET_DEVICE_qualcommax_ipq60xx_DEVICE_jdcloud_re-cs-02_lite=y" >> .config
          make defconfig
          make target/compile -j$(nproc) || make target/compile -j1 V=s
          make package/compile -j$(nproc) || make package/compile -j1 V=s
          make package/index
          
          # 重命名并保存Lite版本
          mkdir -p ../firmware/lite
          cp bin/targets/qualcommax/ipq60xx/*lite* ../firmware/lite/ || true
          
          # 清理并编译Full版本
          rm -rf bin/
          make clean
          sed -i '/_lite=y/d' .config
          make defconfig
          make target/compile -j$(nproc) || make target/compile -j1 V=s
          make package/compile -j$(nproc) || make package/compile -j1 V=s
          make package/index
          
          # 保存Full版本
          mkdir -p ../firmware/full
          cp bin/targets/qualcommax/ipq60xx/* ../firmware/full/ || true

      - name: 收集编译产物 (Collect build artifacts)
        run: |
          cd openwrt
          mkdir -p ../artifacts
          cp .config ../artifacts/config-immortalwrt-ipq60xx.txt
          cp bin/targets/qualcommax/ipq60xx/sha256sums ../artifacts/ || true

      - name: 上传固件 (Upload firmware)
        uses: actions/upload-artifact@v4
        with:
          name: immortalwrt-ipq60xx-firmware
          path: firmware/

      - name: 上传配置文件 (Upload config files)
        uses: actions/upload-artifact@v4
        with:
          name: immortalwrt-ipq60xx-config
          path: artifacts/

      - name: 发布Release (Create Release)
        uses: softprops/action-gh-release@v1
        if: github.event_name == 'workflow_dispatch'
        with:
          tag_name: immortalwrt-ipq60xx-${{ github.run_number }}${{ github.event.inputs.version_suffix }}
          name: ImmortalWrt IPQ60XX 固件 ${{ github.run_number }}${{ github.event.inputs.version_suffix }}
          files: |
            firmware/**/*.trx
            firmware/**/*.bin
            firmware/**/*.img
            artifacts/*

#!/bin/bash
# scripts/disk-monitor.sh

# 磁盘空间监控脚本

while true; do
  # 检查根分区可用空间（GB）
  available_space=$(df / | awk 'NR==2 {print $4}')
  available_gb=$((available_space / 1024 / 1024))
  
  echo "当前可用磁盘空间: ${available_gb}GB"
  
  # 如果可用空间小于10GB，发出警告
  if [ $available_gb -lt 10 ]; then
    echo "警告：磁盘空间不足10GB！当前可用: ${available_gb}GB"
    # 可以在这里添加更多警告逻辑，如发送通知等
  fi
  
  # 每15分钟检查一次
  sleep 900
done

#!/bin/bash
# scripts/build-error-handler.sh
# 编译错误处理脚本

handle_error() {
  local exit_code=$?
  local error_msg=$1
  
  echo "编译过程中发生错误: $error_msg"
  echo "退出码: $exit_code"
  
  # 保存错误日志
  dmesg > build-error-log.txt 2>&1
  echo "错误信息:" >> build-error-log.txt
  echo "$error_msg" >> build-error-log.txt
  
  # 上传错误日志到GitHub Release
  # 注意：需要在工作流中配置GitHub Token
  if [ -n "$GITHUB_TOKEN" ]; then
    echo "上传错误日志到Release..."
    # 这里可以添加上传逻辑
  fi
  
  exit $exit_code
}

# 设置错误处理
trap 'handle_error "未知错误"' ERR

#!/bin/bash
# scripts/common.sh
# 通用函数脚本

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 打印带颜色的信息
print_info() {
  echo -e "${GREEN}[INFO]${NC} $1"
}

print_warning() {
  echo -e "${YELLOW}[WARNING]${NC} $1"
}

print_error() {
  echo -e "${RED}[ERROR]${NC} $1"
}

# 检查命令执行结果
check_result() {
  if [ $? -ne 0 ]; then
    print_error "$1"
    exit 1
  fi
}

# 显示进度条
show_progress() {
  local duration=$1
  local interval=1
  local elapsed=0
  
  while [ $elapsed -lt $duration ]; do
    printf "\r进度: %d/%d 秒" $elapsed $duration
    sleep $interval
    elapsed=$((elapsed + interval))
  done
  printf "\r进度: 完成!          \n"
}
